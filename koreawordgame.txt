local BUTTON_OPACITY_FILE = "끝말봇_button_opacity.txt"
local buttonOpacity = 1

pcall(function()
        if isfile and isfile(BUTTON_OPACITY_FILE) then
                local savedOpacity = tonumber(readfile(BUTTON_OPACITY_FILE))
                if savedOpacity then
                        buttonOpacity = math.clamp(savedOpacity, 0, 1)
                end
        end
end)

local function isOpacityControl(obj)
        local ok, value = pcall(function()
                return obj:GetAttribute("OpacityControl")
        end)
        return ok and value == true
end

local function markOpacityControl(obj)
        pcall(function()
                obj:SetAttribute("OpacityControl", true)
        end)
        return obj
end

local function applyButtonOpacity()
        local transparency = 1 - math.clamp(buttonOpacity, 0, 1)
        for _, obj in ipairs(gui:GetDescendants()) do
                if obj:IsA("TextButton") and not isOpacityControl(obj) then
                        obj.BackgroundTransparency = transparency
                end
        end
end

gui.DescendantAdded:Connect(function(obj)
        if obj:IsA("TextButton") then
                task.defer(function()
                        if not isOpacityControl(obj) then
                                obj.BackgroundTransparency = 1 - math.clamp(buttonOpacity, 0, 1)
                        end
                end)
        end
end)

local dragging = false
local dragInput
local dragStart
local startPos

local function update(input)
        local delta = input.Position - dragStart
        frame.Position = UDim2.new(
                startPos.X.Scale,
                startPos.X.Offset + delta.X,
                startPos.Y.Scale,
                startPos.Y.Offset + delta.Y
        )
end

frame.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1
                or input.UserInputType == Enum.UserInputType.Touch then

                dragging = true
                dragStart = input.Position
                startPos = frame.Position

                input.Changed:Connect(function()
                        if input.UserInputState == Enum.UserInputState.End then
                                dragging = false
                        end
                end)
        end
end)

frame.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement
                or input.UserInputType == Enum.UserInputType.Touch then
                dragInput = input
        end
end)

UserInputService.InputChanged:Connect(function(input)
        if input == dragInput and dragging then
                update(input)
        end
end)

        if url:sub(1, 10) == "content://" then

-- URL 다운로드 → PNG 저장 → 목록 추가 → 적용
local function applyBg(url)
        bgStatusLbl.TextColor3 = Color3.fromRGB(200, 160, 60)
        bgStatusLbl.Text = "다운로드 중..."
        bgImgCount = bgImgCount + 1
        
        local ext = url:lower():match("%.([^%.]+)$") or "png"
        if ext:find("?") then ext = ext:split("?")[1] end
        local fname = "끝말봇_bg_" .. bgImgCount .. "." .. ext
        
        local ok = pcall(function()
                writefile(fname, fetchBytes(url))
                local asset = getcustomasset(fname)
                bgImage.Visible = false
                bgVideo.Visible = false
                bgVideo:Pause()
                if ext == "mp4" or ext == "webm" or ext == "mov" then
                                        safeLoadVideo(bgVideo, asset)
                else
                        bgImage.Image = asset
                        bgImage.Visible = true
                end
        end)
        if ok then
                bgStatusLbl.TextColor3 = Color3.fromRGB(70, 220, 110)
                bgStatusLbl.Text = "v 저장 & 적용 완료"
                table.insert(bgFiles, fname)
                bgCurrentFname = fname
                saveCurrentFname()
                refreshBgList()
        else
                bgImgCount = bgImgCount - 1   -- 실패 시 카운터 롤백
                bgStatusLbl.TextColor3 = Color3.fromRGB(230, 70, 70)
                bgStatusLbl.Text = "x 실패 — URL 확인 필요"
        end
end

                                table.insert(bgFiles, fname)
                                bgImgCount = n
                                found = true
                                break
                        end
                end
                
                if not found then
                        if n > bgImgCount + 5 then break end
                end
        end
        -- 마지막 적용 파일 복원
        if isfile(BG_CURRENT_FILE) then
                local saved = readfile(BG_CURRENT_FILE):match("^%s*(.-)%s*$")
                if saved and saved ~= "" and isfile(saved) then
                        bgCurrentFname  = saved
                        local asset = getcustomasset(saved)
                        local ext = saved:lower():match("%.([^%.]+)$")
                        
                        bgImage.Visible = false
                        bgVideo.Visible = false
                        bgVideo:Pause()

                        if ext == "mp4" or ext == "webm" or ext == "mov" then
                                        safeLoadVideo(bgVideo, asset)
                        else
                                bgImage.Image = asset
                                bgImage.Visible = true
                        end
                end
        end
end)

        local barX = opacityBar.AbsolutePosition.X

        buttonOpacity = math.clamp((input.Position.X - barX) / barW, 0, 1)

local function beginOpacityDrag(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1
                or input.UserInputType == Enum.UserInputType.Touch then
                opacityDragging = true
                setButtonOpacityFromInput(input, true)
                input.Changed:Connect(function()
                        if input.UserInputState == Enum.UserInputState.End then
                                opacityDragging = false
                                saveButtonOpacity()
                        end
                end)
        end
end

local function firstChar(str)
        for _, c in utf8.codes(str) do
                return utf8.char(c)
        end
end

local function lastChar(str)
        local last
        for _, c in utf8.codes(str) do
                last = utf8.char(c)
        end
        return last
end

local function validWord(w)
        return w
                and utf8.len(w)
                and utf8.len(w) >= 2
                and w:match("^[가-힣]+$")
end

local CHO = {
        "ㄱ","ㄲ","ㄴ","ㄷ","ㄸ","ㄹ",
        "ㅁ","ㅂ","ㅃ","ㅅ","ㅆ","ㅇ",
        "ㅈ","ㅉ","ㅊ","ㅋ","ㅌ","ㅍ","ㅎ"
}

local function getCho(char)
        local code = utf8.codepoint(char)
        if code >= 44032 and code <= 55203 then
                local base = code - 44032
                local cho = math.floor(base / 588) + 1
                return CHO[cho]
        end
        return char
end

-- 두음법칙

local N_TO_O = {[2]=true,[3]=true,[6]=true,[7]=true,[12]=true,[17]=true,[20]=true}
local R_TO_O = {[2]=true,[3]=true,[4]=true,[6]=true,[7]=true,[12]=true,[16]=true,[17]=true,[20]=true}

local function getChoIndex(char)
        local code = utf8.codepoint(char)
        if code >= 44032 and code <= 55203 then
                return math.floor((code - 44032) / 588)
        end
        return nil
end

local function getJungIndex(char)
        local code = utf8.codepoint(char)
        if code >= 44032 and code <= 55203 then
                return math.floor(((code - 44032) % 588) / 28)
        end
        return nil
end

local function replaceChoPart(char, newChoIdx)
        local code = utf8.codepoint(char)
        if code >= 44032 and code <= 55203 then
                local rest = (code - 44032) % 588
                return utf8.char(44032 + newChoIdx * 588 + rest)
        end
        return char
end

local function getDuumVariants(startChar)
        local choIdx  = getChoIndex(startChar)
        local jungIdx = getJungIndex(startChar)

        if not choIdx or not jungIdx then
                return {startChar}
        end

        local variants = {}
        local seen = {}

        local function add(cho)
                local v = replaceChoPart(startChar, cho)
                if not seen[v] then
                        seen[v] = true
                        table.insert(variants, v)
                end
        end

        if choIdx == 2 then
                if N_TO_O[jungIdx] then
                        add(11)
                        add(2)
                else
                        add(2)
                end
        elseif choIdx == 5 then
                if R_TO_O[jungIdx] then
                        add(11)
                        add(5)
                elseif jungIdx == 5 then
                        add(5)
                else
                        add(2)
                        add(5)
                end
        else
                add(choIdx)
        end

        return variants
end

                        elseif searchType == "NORMAL" then ok = entry.typePri >= 3

                end

                table.sort(filtered, function(a, b)
                        if searchType == "ALL" then
                                if a.typePri ~= b.typePri then return a.typePri < b.typePri end
                                if a.posPri  ~= b.posPri  then return a.posPri  < b.posPri  end
                                if a.len     ~= b.len     then return a.len     > b.len     end
                                return a.word < b.word
                        elseif searchType == "LONG" then
                                if a.len ~= b.len then return a.len > b.len end
                                return a.word < b.word
                        elseif searchType == "ATTACK" then
                                if a.typePri ~= b.typePri then return a.typePri < b.typePri end
                                if a.len ~= b.len then return a.len > b.len end
                                return a.word < b.word
                        else
                                if a.posPri ~= b.posPri then return a.posPri < b.posPri end
                                return a.word < b.word
                        end
                end)

                allResults = filtered

                if #allResults == 0 then
                        searchCountLbl.Text = "결과 없음"
                else
                        searchCountLbl.Text = "총 " .. #allResults .. "개"
                end

        end

        -- 데이터

        local usedWords   = {}
        local recentWords = {}
        local lastStartChar = nil

        local function isUsed(word)
                return usedWords[word] == true
        end

        local function addRecent(word)
                table.insert(recentWords, 1, word)
                if #recentWords > 20 then
                        table.remove(recentWords)
                end
        end

        local function wasRecent(word)
                for _, v in ipairs(recentWords) do
                        if v == word then return true end
                end
                return false
        end

        local function shuffle(tbl)
                for i = #tbl, 2, -1 do
                        local j = math.random(i)
                        tbl[i], tbl[j] = tbl[j], tbl[i]
                end
        end

        local function pickOne(candidates)
                if #candidates <= 0 then return nil end
                shuffle(candidates)
                local top = {}
                for i = 1, math.min(5, #candidates) do
                        table.insert(top, candidates[i])
                end
                return top[math.random(1, #top)]
        end

        -- 한방

        local function findByScore(startChar, filterFn, topN)
                topN = topN or 3
                local candidates = {}
                local seenW = {}

                local function tryList(list)
                        if not list then return end
                        for _, word in ipairs(list) do
                                if isDeleted(word) then continue end
                                if not seenW[word] and not isUsed(word) and not wasRecent(word) then
                                        seenW[word] = true
                                        if not filterFn or filterFn(word) then
                                                table.insert(candidates, word)
                                        end
                                end
                        end
                end

                tryList(dictionary[startChar])
                if #candidates == 0 then
                        for _, variant in ipairs(getDuumVariants(startChar)) do
                                if variant ~= startChar then
                                        tryList(dictionary[variant])
                                end
                        end
                end

                if #candidates == 0 then return nil end
                table.sort(candidates, function(a, b)
                        return (wordScore[a] or 0) > (wordScore[b] or 0)
                end)
                local selected = candidates[math.random(1, math.min(topN, #candidates))]
                if selected then addRecent(selected) end
                return selected
        end

        local function findBestWord(startChar)
                return findByScore(startChar, function(w)
                        return (wordScore[w] or 0) >= 100000
                end, 3)
        end

        ---

        -- 미니맥스 (Negamax + 알파베타, depth-2)
        -- 내 후보 100개 선별 후 depth-2 수읽기로 한방까지 가는 최적 루트 탐색

        -- GUI 박스에서 실시간으로 읽는 설정값
        local function readMMSettings()
                local depth   = math.max(1, math.min(10, math.floor(tonumber(mmDepthBox.Text)  or 2)))
                local sample  = math.max(1, math.min(200, math.floor(tonumber(mmSampleBox.Text) or 10)))
                local timeLim = math.max(0, tonumber(mmTimeBox.Text) or 0)
                mmDepthBox.Text  = tostring(depth)
                mmSampleBox.Text = tostring(sample)
                mmTimeBox.Text   = tostring(timeLim)
                return depth, sample, sample * 10, timeLim
                -- 반환: depth, MINIMAX_SAMPLE, MY_CANDIDATES(샘플의 10배), timeLimitSec
        end

        local MINIMAX_DEPTH   = 2    -- 아래 findMiniMaxWord에서 매 탐색마다 갱신
        local MINIMAX_SAMPLE  = 10
        local MY_CANDIDATES   = 100

        -- 탐색 마감 시각 (tick 기준), 0 = 제한 없음
        local searchDeadline  = 0

        -- startChar 기준 후보 단어 최대 maxN개 수집 (점수 내림차순)
        -- extraFilter: function(word)→bool  추가 필터링 (nil = 없음)
        local function collectCandidates(startChar, maxN, extraFilter)
                maxN = maxN or MINIMAX_SAMPLE
                local candidates = {}
                local seenW = {}

                local function tryList(list)
                        if not list then return end
                        for _, word in ipairs(list) do
                                if isDeleted(word) then continue end
                                if not seenW[word] and not isUsed(word) and not wasRecent(word) then
                                        if not extraFilter or extraFilter(word) then
                                                seenW[word] = true
                                                table.insert(candidates, word)
                                        end
                                end
                        end
                end

                -- 정확한 글자 우선
                tryList(dictionary[startChar])
                -- 부족하면 두음법칙 변형 fallback
                if #candidates < maxN then
                        for _, variant in ipairs(getDuumVariants(startChar)) do
                                if variant ~= startChar then
                                        tryList(dictionary[variant])
                                        if #candidates >= maxN * 3 then break end
                                end
                        end
                end

                -- 점수 내림차순 정렬
                table.sort(candidates, function(a, b)
                        return (wordScore[a] or 0) > (wordScore[b] or 0)
                end)

                local result = {}
                for i = 1, math.min(maxN, #candidates) do
                        result[i] = candidates[i]
                end
                return result
        end

        -- ───────────────────────────────────────────
        -- Negamax + 알파베타 가지치기
        --
        -- 반환값: 현재 플레이어 관점의 점수 (양수 = 유리)
        -- char   : 현재 플레이어가 시작해야 할 글자
        -- depth  : 남은 탐색 깊이 (0이면 리프 평가)
        -- alpha  : 현재 플레이어가 보장한 최솟값 (최대화 하한)
        -- beta   : 상대가 허용하는 최댓값 (가지치기 상한)
        --
        -- 점수 계층 (높을수록 유리):
        --   +999999  즉사 (상대 응수 없음)
        --   +500000~ 한방단어 (score >= 100000)
        --   + 50000~ 2-kill 경로 (score >= 10000)
        --   기타     wordScore 기반
        -- ───────────────────────────────────────────
        local function negamax(char, depth, alpha, beta)
                -- 탐색 시간 초과 체크 (searchDeadline == 0 이면 무제한)
                if searchDeadline > 0 and tick() >= searchDeadline then
                        return 0  -- 시간 초과 시 중립값 반환 → 상위에서 현재까지 최선값 유지
                end

                local cands = collectCandidates(char, MINIMAX_SAMPLE)

                -- 응수 불가 = 현재 플레이어 패배
                if #cands == 0 then
                        return -999999
                end

                -- 리프 노드: 최고점 단어 점수로 평가
                if depth == 0 then
                        local best = -math.huge
                        for _, w in ipairs(cands) do
                                local s = wordScore[w] or 0
                                if s > best then best = s end
                        end
                        return best
                end

                local best = -math.huge

                for _, word in ipairs(cands) do
                        -- 루프 안에서도 시간 초과 체크
                        if searchDeadline > 0 and tick() >= searchDeadline then break end

                        local ws = wordScore[word] or 0
                        local lc = lastChar(word)

                        -- 상대 응수 가능 여부 빠른 체크
                        local opFirst = collectCandidates(lc, 1)
                        local val

                        if #opFirst == 0 then
                                -- 즉사 (한방단어) → 최고 점수, 탐색 즉시 종료
                                val = 999999
                        else
                                -- 상대 관점 negamax → negate 해서 내 관점으로 변환
                                val = ws + (- negamax(lc, depth - 1, -beta, -alpha))
                        end

                        if val > best then best = val end
                        if best > alpha then alpha = best end
                        -- 알파베타 가지치기
                        if alpha >= beta then break end
                end

                return best
        end

        -- 내 후보 100개에서 depth-2 negamax로 최적 단어 선택
        -- rootFilter: 첫 수 후보를 추가 필터링하는 함수 (nil = 필터 없음)
        -- 노멀 모드: 첫 수에서 즉사단어 제외 → 상대가 받아치기 힘든 세팅 단어 우선
        -- 이후 탐색 트리에서 두 번째 수로 한방단어 도달 경로를 탐색
        local function findMiniMaxWord(startChar, rootFilter)
                -- 탐색 시작 직전에 GUI 박스에서 설정값 읽기
                local depth, sample, myCandN, timeLim = readMMSettings()
                MINIMAX_DEPTH  = depth
                MINIMAX_SAMPLE = sample
                MY_CANDIDATES  = myCandN

                -- 탐색 마감 시각 설정 (0 = 무제한)
                if timeLim > 0 then
                        searchDeadline = tick() + timeLim
                        status.Text = string.format("미니맥스 탐색중 (깊이%d / 후보%d / %.1f초)", depth, sample, timeLim)
                else
                        searchDeadline = 0
                        status.Text = string.format("미니맥스 탐색중 (깊이%d / 후보%d / 무제한)", depth, sample)
                end

                local myCands = collectCandidates(startChar, MY_CANDIDATES, rootFilter)

                -- rootFilter 적용 후 후보가 없으면 필터 없이 재수집 (안전장치)
                if #myCands == 0 then
                        myCands = collectCandidates(startChar, MY_CANDIDATES)
                end
                if #myCands == 0 then return nil end

                local bestWord = nil
                local bestEval = -math.huge
                local alpha    = -math.huge
                local beta     =  math.huge

                for _, word in ipairs(myCands) do
                        -- 루트 레벨에서도 시간 초과 체크
                        if searchDeadline > 0 and tick() >= searchDeadline then break end

                        local ws = wordScore[word] or 0
                        local lc = lastChar(word)

                        local opFirst = collectCandidates(lc, 1)
                        local eval

                        if #opFirst == 0 then
                                eval = 999999 + ws
                        else
                                eval = ws + (- negamax(lc, MINIMAX_DEPTH - 1, -beta, -alpha))
                        end

                        if eval > bestEval then
                                bestEval = eval
                                bestWord = word
                        end
                        if bestEval > alpha then alpha = bestEval end
                        if alpha >= beta then break end
                end

                if bestWord then addRecent(bestWord) end
                return bestWord
        end

        -- 생성

        local function generateWord(startChar)
                if useMinimax then
                        local rootFilter = nil

                        if mode == "NORMAL" then
                                -- 노멀 + 미니맥스:
                                -- 첫 수에서 즉사단어(한방) 완전 제외
                                -- 대신 상대가 이어받기 힘든 2-kill 세팅 단어 or 끝글자가 나에게 유리한 공격단어 우선
                                -- depth-2 탐색으로 두 번째 수에서 한방 마무리 경로를 자동 탐색
                                rootFilter = function(w)
                                        local s = wordScore[w] or 0
                                        -- 즉사단어(score >= 100000) 제외
                                        -- 필수공격·공격단어(attackBonus 포함) 우선 자동 반영됨 (점수에 포함되어 있음)
                                        return s < 100000
                                end
                        end

                        local mmWord = findMiniMaxWord(startChar, rootFilter)
                        if mmWord then return mmWord end
                end

                if mode == "ONE" then
                        return findBestWord(startChar)
                elseif mode == "LONG" then
                        return findLongestWord(startChar)
                else
                        return findNormalWord(startChar)
                end
        end

        end

                startPos = frame.Position

                input.Changed:Connect(function()

                                dragging = false
                        end
                end)
        end
end)

        end

                for _, ext in ipairs(exts) do
                        local fname = "끝말봇_bg_" .. n .. ext
                        if isfile(fname) then
                                table.insert(bgFiles, fname)
                                bgImgCount = n
                                found = true
                                break
                        end
                end
                
                if not found then
                        if n > bgImgCount + 5 then break end
                end
        end
        -- 마지막 적용 파일 복원
        if isfile(BG_CURRENT_FILE) then
                local saved = readfile(BG_CURRENT_FILE):match("^%s*(.-)%s*$")
                if saved and saved ~= "" and isfile(saved) then
                        bgCurrentFname  = saved
                        local asset = getcustomasset(saved)
                        local ext = saved:lower():match("%.([^%.]+)$")
                        
                        bgImage.Visible = false
                        bgVideo.Visible = false
                        bgVideo:Pause()

                        if ext == "mp4" or ext == "webm" or ext == "mov" then
                                        safeLoadVideo(bgVideo, asset)
                        else
                                bgImage.Image = asset
                                bgImage.Visible = true
                        end
                end
        end
end)

        local barX = opacityBar.AbsolutePosition.X

        if input.UserInputType == Enum.UserInputType.MouseButton1
                or input.UserInputType == Enum.UserInputType.Touch then
                opacityDragging = true
                setButtonOpacityFromInput(input, true)
                input.Changed:Connect(function()
                        if input.UserInputState == Enum.UserInputState.End then
                                opacityDragging = false
                                saveButtonOpacity()
                        end
                end)
        end

local function firstChar(str)
        for _, c in utf8.codes(str) do
                return utf8.char(c)
        end
end

local function lastChar(str)
        local last
        for _, c in utf8.codes(str) do
                last = utf8.char(c)
        end
        return last
end

local function validWord(w)
        return w
                and utf8.len(w)
                and utf8.len(w) >= 2
                and w:match("^[가-힣]+$")
end

local CHO = {
        "ㄱ","ㄲ","ㄴ","ㄷ","ㄸ","ㄹ",
        "ㅁ","ㅂ","ㅃ","ㅅ","ㅆ","ㅇ",
        "ㅈ","ㅉ","ㅊ","ㅋ","ㅌ","ㅍ","ㅎ"
}

local function getCho(char)
        local code = utf8.codepoint(char)
        if code >= 44032 and code <= 55203 then
                local base = code - 44032
                local cho = math.floor(base / 588) + 1
                return CHO[cho]
        end
        return char
end

-- 두음법칙

local N_TO_O = {[2]=true,[3]=true,[6]=true,[7]=true,[12]=true,[17]=true,[20]=true}
local R_TO_O = {[2]=true,[3]=true,[4]=true,[6]=true,[7]=true,[12]=true,[16]=true,[17]=true,[20]=true}

local function getChoIndex(char)
        local code = utf8.codepoint(char)
        if code >= 44032 and code <= 55203 then
                return math.floor((code - 44032) / 588)
        end
        return nil
end

local function getJungIndex(char)
        local code = utf8.codepoint(char)
        if code >= 44032 and code <= 55203 then
                return math.floor(((code - 44032) % 588) / 28)
        end
        return nil
end

local function replaceChoPart(char, newChoIdx)
        local code = utf8.codepoint(char)
        if code >= 44032 and code <= 55203 then
                local rest = (code - 44032) % 588
                return utf8.char(44032 + newChoIdx * 588 + rest)
        end
        return char
end

local function getDuumVariants(startChar)
        local choIdx  = getChoIndex(startChar)
        local jungIdx = getJungIndex(startChar)

        if not choIdx or not jungIdx then
                return {startChar}
        end

        local variants = {}
        local seen = {}

        local function add(cho)
                local v = replaceChoPart(startChar, cho)
                if not seen[v] then
                        seen[v] = true
                        table.insert(variants, v)
                end
        end

        if choIdx == 2 then
                if N_TO_O[jungIdx] then
                        add(11)
                        add(2)
                else
                        add(2)
                end
        elseif choIdx == 5 then
                if R_TO_O[jungIdx] then
                        add(11)
                        add(5)
                elseif jungIdx == 5 then
                        add(5)
                else
                        add(2)
                        add(5)
                end
        else
                add(choIdx)
        end

        return variants
end

                                                        isKill   = not hasWordStarting(lastChar(word)),
                                                        len      = wLen,
                                                        posPri   = posPri,
                                                        typePri  = typePri,
                                                })
                                                count += 1
                                                if count >= SEARCH_RESULT_CAP then break end
                                        end
                                end
                                if count >= SEARCH_RESULT_CAP then break end
                        end
                        if count >= SEARCH_RESULT_CAP then break end
                        if count % 1000 == 0 then
                                task.wait()
                                searchCountLbl.Text = "검색 중... " .. count .. "개"
                        end
                end

                local filtered = {}
                for _, entry in ipairs(raw) do
                        local ok = false
                        if     searchType == "ALL"    then ok = true
                        elseif searchType == "ATTACK" then ok = entry.isKill
                        elseif searchType == "LONG"   then ok = entry.len >= 5
                        elseif searchType == "NORMAL" then ok = entry.typePri >= 3
                        end
                        if ok then table.insert(filtered, entry) end
                end

                table.sort(filtered, function(a, b)
                        if searchType == "ALL" then
                                if a.typePri ~= b.typePri then return a.typePri < b.typePri end
                                if a.posPri  ~= b.posPri  then return a.posPri  < b.posPri  end
                                if a.len     ~= b.len     then return a.len     > b.len     end
                                return a.word < b.word
                        elseif searchType == "LONG" then
                                if a.len ~= b.len then return a.len > b.len end
                                return a.word < b.word
                        elseif searchType == "ATTACK" then
                                if a.typePri ~= b.typePri then return a.typePri < b.typePri end
                                if a.len ~= b.len then return a.len > b.len end
                                return a.word < b.word
                        else
                                if a.posPri ~= b.posPri then return a.posPri < b.posPri end
                                return a.word < b.word
                        end
                end)

                allResults = filtered

                if #allResults == 0 then
                        searchCountLbl.Text = "결과 없음"
                else
                        searchCountLbl.Text = "총 " .. #allResults .. "개"
                end

                renderAll()
        end

        -- 데이터

        local usedWords   = {}
        local recentWords = {}
        local lastStartChar = nil

        local function isUsed(word)
                return usedWords[word] == true
        end

        local function addRecent(word)
                table.insert(recentWords, 1, word)
                if #recentWords > 20 then
                        table.remove(recentWords)
                end
        end

        local function wasRecent(word)
                for _, v in ipairs(recentWords) do
                        if v == word then return true end
                end
                return false
        end

        local function shuffle(tbl)
                for i = #tbl, 2, -1 do
                        local j = math.random(i)
                        tbl[i], tbl[j] = tbl[j], tbl[i]
                end
        end

        local function pickOne(candidates)
                if #candidates <= 0 then return nil end
                shuffle(candidates)
                local top = {}
                for i = 1, math.min(5, #candidates) do
                        table.insert(top, candidates[i])
                end
                return top[math.random(1, #top)]
        end

        -- 한방

        local function findByScore(startChar, filterFn, topN)
                topN = topN or 3
                local candidates = {}
                local seenW = {}

                local function tryList(list)
                        if not list then return end
                        for _, word in ipairs(list) do
                                if isDeleted(word) then continue end
                                if not seenW[word] and not isUsed(word) and not wasRecent(word) then
                                        seenW[word] = true
                                        if not filterFn or filterFn(word) then
                                                table.insert(candidates, word)
                                        end
                                end
                        end
                end

                tryList(dictionary[startChar])
                if #candidates == 0 then
                        for _, variant in ipairs(getDuumVariants(startChar)) do
                                if variant ~= startChar then
                                        tryList(dictionary[variant])
                                end
                        end
                end

                if #candidates == 0 then return nil end
                table.sort(candidates, function(a, b)
                        return (wordScore[a] or 0) > (wordScore[b] or 0)
                end)
                local selected = candidates[math.random(1, math.min(topN, #candidates))]
                if selected then addRecent(selected) end
                return selected
        end

        local function findBestWord(startChar)
                return findByScore(startChar, function(w)
                        return (wordScore[w] or 0) >= 100000
                end, 3)
        end

                local seenW = {}

                local function tryList(list)

        ---

        -- 미니맥스 (Negamax + 알파베타, depth-2)
        -- 내 후보 100개 선별 후 depth-2 수읽기로 한방까지 가는 최적 루트 탐색

        -- GUI 박스에서 실시간으로 읽는 설정값
        local function readMMSettings()
                local depth   = math.max(1, math.min(10, math.floor(tonumber(mmDepthBox.Text)  or 2)))
                local sample  = math.max(1, math.min(200, math.floor(tonumber(mmSampleBox.Text) or 10)))
                local timeLim = math.max(0, tonumber(mmTimeBox.Text) or 0)
                mmDepthBox.Text  = tostring(depth)
                mmSampleBox.Text = tostring(sample)
                mmTimeBox.Text   = tostring(timeLim)
                return depth, sample, sample * 10, timeLim
                -- 반환: depth, MINIMAX_SAMPLE, MY_CANDIDATES(샘플의 10배), timeLimitSec
        end

        local MINIMAX_DEPTH   = 2    -- 아래 findMiniMaxWord에서 매 탐색마다 갱신
        local MINIMAX_SAMPLE  = 10
        local MY_CANDIDATES   = 100

        -- 탐색 마감 시각 (tick 기준), 0 = 제한 없음
        local searchDeadline  = 0

        -- startChar 기준 후보 단어 최대 maxN개 수집 (점수 내림차순)
        -- extraFilter: function(word)→bool  추가 필터링 (nil = 없음)
        local function collectCandidates(startChar, maxN, extraFilter)
                maxN = maxN or MINIMAX_SAMPLE
                local candidates = {}
                local seenW = {}

                local function tryList(list)
                        if not list then return end
                        for _, word in ipairs(list) do
                                if isDeleted(word) then continue end
                                if not seenW[word] and not isUsed(word) and not wasRecent(word) then
                                        if not extraFilter or extraFilter(word) then
                                                seenW[word] = true
                                                table.insert(candidates, word)
                                        end
                                end
                        end
                end

                -- 정확한 글자 우선
                tryList(dictionary[startChar])
                -- 부족하면 두음법칙 변형 fallback
                if #candidates < maxN then
                        for _, variant in ipairs(getDuumVariants(startChar)) do
                                if variant ~= startChar then
                                        tryList(dictionary[variant])
                                        if #candidates >= maxN * 3 then break end
                                end
                        end
                end

                -- 점수 내림차순 정렬
                table.sort(candidates, function(a, b)
                        return (wordScore[a] or 0) > (wordScore[b] or 0)
                end)

                local result = {}
                for i = 1, math.min(maxN, #candidates) do
                        result[i] = candidates[i]
                end
                return result
        end

        -- ───────────────────────────────────────────
        -- Negamax + 알파베타 가지치기
        --
        -- 반환값: 현재 플레이어 관점의 점수 (양수 = 유리)
        -- char   : 현재 플레이어가 시작해야 할 글자
        -- depth  : 남은 탐색 깊이 (0이면 리프 평가)
        -- alpha  : 현재 플레이어가 보장한 최솟값 (최대화 하한)
        -- beta   : 상대가 허용하는 최댓값 (가지치기 상한)
        --
        -- 점수 계층 (높을수록 유리):
        --   +999999  즉사 (상대 응수 없음)
        --   +500000~ 한방단어 (score >= 100000)
        --   + 50000~ 2-kill 경로 (score >= 10000)
        --   기타     wordScore 기반
        -- ───────────────────────────────────────────
        local function negamax(char, depth, alpha, beta)
                -- 탐색 시간 초과 체크 (searchDeadline == 0 이면 무제한)
                if searchDeadline > 0 and tick() >= searchDeadline then
                        return 0  -- 시간 초과 시 중립값 반환 → 상위에서 현재까지 최선값 유지
                end

                local cands = collectCandidates(char, MINIMAX_SAMPLE)

                -- 응수 불가 = 현재 플레이어 패배
                if #cands == 0 then
                        return -999999
                end

                -- 리프 노드: 최고점 단어 점수로 평가
                if depth == 0 then
                        local best = -math.huge
                        for _, w in ipairs(cands) do
                                local s = wordScore[w] or 0
                                if s > best then best = s end
                        end
                        return best
                end

                local best = -math.huge

                for _, word in ipairs(cands) do
                        -- 루프 안에서도 시간 초과 체크
                        if searchDeadline > 0 and tick() >= searchDeadline then break end

                        local ws = wordScore[word] or 0
                        local lc = lastChar(word)

                        -- 상대 응수 가능 여부 빠른 체크
                        local opFirst = collectCandidates(lc, 1)
                        local val

                        if #opFirst == 0 then
                                -- 즉사 (한방단어) → 최고 점수, 탐색 즉시 종료
                                val = 999999
                        else
                                -- 상대 관점 negamax → negate 해서 내 관점으로 변환
                                val = ws + (- negamax(lc, depth - 1, -beta, -alpha))
                        end

                        if val > best then best = val end
                        if best > alpha then alpha = best end
                        -- 알파베타 가지치기
                        if alpha >= beta then break end
                end

                return best
        end

        -- 내 후보 100개에서 depth-2 negamax로 최적 단어 선택
        -- rootFilter: 첫 수 후보를 추가 필터링하는 함수 (nil = 필터 없음)
        -- 노멀 모드: 첫 수에서 즉사단어 제외 → 상대가 받아치기 힘든 세팅 단어 우선
        -- 이후 탐색 트리에서 두 번째 수로 한방단어 도달 경로를 탐색
        local function findMiniMaxWord(startChar, rootFilter)
                -- 탐색 시작 직전에 GUI 박스에서 설정값 읽기
                local depth, sample, myCandN, timeLim = readMMSettings()
                MINIMAX_DEPTH  = depth
                MINIMAX_SAMPLE = sample
                MY_CANDIDATES  = myCandN

                -- 탐색 마감 시각 설정 (0 = 무제한)
                if timeLim > 0 then
                        searchDeadline = tick() + timeLim
                        status.Text = string.format("미니맥스 탐색중 (깊이%d / 후보%d / %.1f초)", depth, sample, timeLim)
                else
                        searchDeadline = 0
                        status.Text = string.format("미니맥스 탐색중 (깊이%d / 후보%d / 무제한)", depth, sample)
                end

                local myCands = collectCandidates(startChar, MY_CANDIDATES, rootFilter)

                -- rootFilter 적용 후 후보가 없으면 필터 없이 재수집 (안전장치)
                if #myCands == 0 then
                        myCands = collectCandidates(startChar, MY_CANDIDATES)
                end
                if #myCands == 0 then return nil end

                local bestWord = nil
                local bestEval = -math.huge
                local alpha    = -math.huge
                local beta     =  math.huge

                for _, word in ipairs(myCands) do
                        -- 루트 레벨에서도 시간 초과 체크
                        if searchDeadline > 0 and tick() >= searchDeadline then break end

                        local ws = wordScore[word] or 0
                        local lc = lastChar(word)

                        local opFirst = collectCandidates(lc, 1)
                        local eval

                        if #opFirst == 0 then
                                eval = 999999 + ws
                        else
                                eval = ws + (- negamax(lc, MINIMAX_DEPTH - 1, -beta, -alpha))
                        end

                        if eval > bestEval then
                                bestEval = eval
                                bestWord = word
                        end
                        if bestEval > alpha then alpha = bestEval end
                        if alpha >= beta then break end
                end

                if bestWord then addRecent(bestWord) end
                return bestWord
        end

        -- 생성

        local function generateWord(startChar)
                if useMinimax then
                        local rootFilter = nil

                        if mode == "NORMAL" then
                                -- 노멀 + 미니맥스:
                                -- 첫 수에서 즉사단어(한방) 완전 제외
                                -- 대신 상대가 이어받기 힘든 2-kill 세팅 단어 or 끝글자가 나에게 유리한 공격단어 우선
                                -- depth-2 탐색으로 두 번째 수에서 한방 마무리 경로를 자동 탐색
                                rootFilter = function(w)
                                        local s = wordScore[w] or 0
                                        -- 즉사단어(score >= 100000) 제외
                                        -- 필수공격·공격단어(attackBonus 포함) 우선 자동 반영됨 (점수에 포함되어 있음)
                                        return s < 100000
                                end
                        end

                        local mmWord = findMiniMaxWord(startChar, rootFilter)
                        if mmWord then return mmWord end
                end

                if mode == "ONE" then
                        return findBestWord(startChar)
                elseif mode == "LONG" then
                        return findLongestWord(startChar)
                else
                        return findNormalWord(startChar)
                end
        end

                end
        end

        task.spawn(function()
                while true do
                        task.wait(0.2)

                        if not trollMode or isTrolling then continue end

                        local ok, turner = pcall(getCurrentTurner)
                        if not ok or not turner then
                                lastTrollTurner = nil
                                continue
                        end

                        -- 같은 사람 턴이면 스킵 (이미 이번 턴에 답했음)
                        if turner == lastTrollTurner then continue end

                        lastTrollTurner = turner
                        isTrolling = true

                        task.spawn(function()
                                doAutoInput()
                                isTrolling = false
                        end)
                end
        end)


end
